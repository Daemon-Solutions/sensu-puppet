CONFIG = 
Proc.new do
  newparam(:chunk_size, :parent => PuppetX::Sensu::IntegerProperty) do
    desc "sensuctl chunk-size"
    defaultto(100)
  end

  define_method(:generate) do
    provider_class = self.class.provider(:sensuctl)
    provider_class.chunk_size = self[:chunk_size]
    []
  end
end
